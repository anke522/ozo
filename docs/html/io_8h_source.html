76 template <typename T, typename = Require<Connection<T>>>
77 inline error_code pq_assign_socket(T& conn) {
78  int fd = PQsocket(get_native_handle(conn));
79  if (fd == -1) {
80  return error::pq_socket_failed;
81  }
82 
83  int new_fd = dup(fd);
84  if (new_fd == -1) {
85  set_error_context(conn, "error while dup(fd) for socket stream");
86  return error_code{errno, boost::system::generic_category()};
87  }
88 
89  error_code ec;
90  get_socket(conn).assign(new_fd, ec);
91 
92  if (ec) {
93  set_error_context(conn, "assign socket failed");
94  }
95  return ec;
96 }
