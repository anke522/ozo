89 template <typename Context, typename BinaryQuery>
90 struct async_send_query_params_op {
91  Context ctx_;
92  BinaryQuery query_;
93 
94  void perform() {
95  decltype(auto) conn = get_connection(ctx_);
96  if (auto ec = set_nonblocking(conn)) {
97  return done(ctx_, ec);
98  }
99  //In the nonblocking state, calls to PQsendQuery, PQputline,
100  //PQputnbytes, PQputCopyData, and PQendcopy will not block
101  //but instead return an error if they need to be called again.
102  while (!send_query_params(conn, query_));
103  post(ctx_, *this);
104  }
105 
106  void operator () (error_code ec = error_code{}, std::size_t = 0) {
107  // if data has been flushed or error has been set by
108  // read operation no write opertion handling is needed
109  // anymore.
110  if (get_query_state(ctx_) != query_state::send_in_progress) {
111  return;
112  }
113 
114  // In case of write operation error - finish the request
115  // with error.
116  if (ec) {
117  return done(ctx_, ec);
118  }
119 
120  // Trying to flush output one more time according to the
121  // documentation
122  switch (flush_output(get_connection(ctx_))) {
123  case query_state::error:
124  done(ctx_, error::pg_flush_failed);
125  break;
126  case query_state::send_in_progress:
127  write_poll(ctx_, *this);
128  break;
129  case query_state::send_finish:
130  set_query_state(ctx_, query_state::send_finish);
131  break;
132  }
133  }

164 template <typename Context, typename ResultProcessor>
165 struct async_get_result_op : boost::asio::coroutine {
166  Context ctx_;
167  ResultProcessor process_;
168 
169  async_get_result_op(Context ctx, ResultProcessor process)
170  : ctx_(ctx), process_(process) {}
171 
172  void perform() {
173  post(ctx_, *this);
174  }
175 
176  void operator() (error_code ec = error_code{}, std::size_t = 0) {
177  // In case when query error state has been set by send query params
178  // operation skip handle and do nothing more.
179  if (get_query_state(ctx_) == query_state::error) {
180  return;
181  }
182 
183  if (ec) {
184  // Bad descriptor error can occur here if the connection
185  // has been closed by user during processing.
186  if (ec == asio::error::bad_descriptor) {
187  ec = asio::error::operation_aborted;
188  }
189  return done(ctx_, ec);
190  }
191 
192  reenter(*this) {
193  while (is_busy(get_connection(ctx_))) {
194  yield read_poll(ctx_, *this);
195  if (auto err = consume_input(get_connection(ctx_))) {
196  return done(ctx_, err);
197  }
198  }
199 
200  if (auto res = get_result(get_connection(ctx_))) {
201  const auto status = result_status(*res);
202  switch (status) {
203  case PGRES_SINGLE_TUPLE:
204  process_and_done(std::move(res));
205  return;
206  case PGRES_TUPLES_OK:
207  process_and_done(std::move(res));
208  consume_result(get_connection(ctx_));
209  return;
210  case PGRES_COMMAND_OK:
211  done(ctx_);
212  consume_result(get_connection(ctx_));
213  return;
214  case PGRES_BAD_RESPONSE:
215  done(ctx_, error::result_status_bad_response);
216  consume_result(get_connection(ctx_));
217  return;
218  case PGRES_EMPTY_QUERY:
219  done(ctx_, error::result_status_empty_query);
220  consume_result(get_connection(ctx_));
221  return;
222  case PGRES_FATAL_ERROR:
223  done(ctx_, result_error(*res));
224  consume_result(get_connection(ctx_));
225  return;
226  case PGRES_COPY_OUT:
227  case PGRES_COPY_IN:
228  case PGRES_COPY_BOTH:
229  case PGRES_NONFATAL_ERROR:
230  break;
231  }
232  set_error_context(get_connection(ctx_), get_result_status_name(status));
233  done(ctx_, error::result_status_unexpected);
234  consume_result(get_connection(ctx_));
235  } else {
236  done(ctx_);
237  }
238  }
239  }
