85 template <typename Context>
86 struct async_connect_op {
87  Context context;
88 
89  void done(error_code ec = error_code {}) {
90  get_timer(context).cancel();
91 
92  get_io_context(get_connection(context))
93  .post(detail::bind(std::move(get_handler(context)), std::move(ec), std::move(get_connection(context))));
94  }
95 
96  void perform(const std::string& conninfo, const time_traits::duration& timeout) {
97  if (error_code ec = start_connection(get_connection(context), conninfo)) {
98  return done(ec);
99  }
100 
101  if (connection_bad(get_connection(context))) {
102  return done(error::pq_connection_status_bad);
103  }
104 
105  if (error_code ec = assign_socket(get_connection(context))) {
106  return done(ec);
107  }
108 
109  get_timer(context).expires_after(timeout);
110  get_timer(context).async_wait(bind_executor(get_executor(context),
111  make_timeout_handler(get_socket(get_connection(context)))));
112 
113  return write_poll(get_connection(context), bind_executor(get_executor(context), *this));
114  }
115 
116  void operator () (error_code ec, std::size_t = 0) {
117  if (ec) {
118  set_error_context(get_connection(context), "error while connection polling");
119  return done(ec);
120  }
121 
122  switch (connect_poll(get_connection(context))) {
123  case PGRES_POLLING_OK:
124  return done();
125 
126  case PGRES_POLLING_WRITING:
127  return write_poll(get_connection(context), std::move(*this));
128 
129  case PGRES_POLLING_READING:
130  return read_poll(get_connection(context), std::move(*this));
131 
132  case PGRES_POLLING_FAILED:
133  case PGRES_POLLING_ACTIVE:
134  break;
135  }
136 
137  done(error::pq_connect_poll_failed);
138  }
